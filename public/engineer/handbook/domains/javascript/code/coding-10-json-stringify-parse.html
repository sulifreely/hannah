编程 10（挑战）：手写 JSON.stringify 与 JSON.parse ​

题目 ​

不使用原生 JSON.stringify / JSON.parse，实现两个函数：

  1. myStringify(value, replacer, space) 把 JS 值序列化成 JSON 字符串；
  2. myParse(text) 把 JSON 字符串反序列化成 JS 值。

支持基本数据类型、数组、对象、嵌套；处理特殊值（undefined、NaN、Infinity、循环引用）；不需要完全对齐 ECMA-404 全部边界，但要抓住核心行为。

参考答案 ​

myStringify ​

js
function myStringify(value, replacer, space) {
  const indent = typeof space === 'number' ? ' '.repeat(Math.min(space, 10))
              : typeof space === 'string' ? space.slice(0, 10)
              : '';
  const seen = new WeakSet();

  function applyReplacer(key, val) {
    if (typeof replacer === 'function') return replacer(key, val);
    return val;
  }

  function serialize(val, key, depth) {
    val = applyReplacer(key, val);

    // 特殊：null 直接返回 'null'
    if (val === null) return 'null';

    // 处理对象上的 toJSON
    if (val && typeof val.toJSON === 'function') {
      val = val.toJSON(key);
    }

    switch (typeof val) {
      case 'number':
        return Number.isFinite(val) ? String(val) : 'null';   // NaN/Infinity → null
      case 'boolean':
        return String(val);
      case 'string':
        return quote(val);
      case 'undefined':
      case 'function':
      case 'symbol':
        return undefined;   // 顶层返回 undefined，父层根据情境处理
      case 'bigint':
        throw new TypeError('Do not know how to serialize a BigInt');
      case 'object':
        if (seen.has(val)) throw new TypeError('Converting circular structure to JSON');
        seen.add(val);

        const nextIndent = indent ? '\n' + indent.repeat(depth + 1) : '';
        const closingIndent = indent ? '\n' + indent.repeat(depth) : '';
        const separator = indent ? ': ' : ':';

        let out;
        if (Array.isArray(val)) {
          if (val.length === 0) out = '[]';
          else {
            const items = val.map((v, i) => {
              const s = serialize(v, String(i), depth + 1);
              return s === undefined ? 'null' : s;   // 数组里 undefined 变 null
            });
            out = `[${nextIndent}${items.join(',' + nextIndent)}${closingIndent}]`;
          }
        } else {
          const entries = [];
          for (const k of Object.keys(val)) {
            const s = serialize(val[k], k, depth + 1);
            if (s === undefined) continue;   // 对象里 undefined/function/symbol 被跳过
            entries.push(`${quote(k)}${separator}${s}`);
          }
          if (entries.length === 0) out = '{}';
          else out = `{${nextIndent}${entries.join(',' + nextIndent)}${closingIndent}}`;
        }
        seen.delete(val);
        return out;
    }
  }

  function quote(s) {
    // 转义控制字符
    const escapes = { '"': '\\"', '\\': '\\\\', '\n': '\\n', '\r': '\\r', '\t': '\\t', '\b': '\\b', '\f': '\\f' };
    let out = '"';
    for (const c of s) {
      if (escapes[c]) out += escapes[c];
      else if (c.charCodeAt(0) < 0x20) out += '\\u' + c.charCodeAt(0).toString(16).padStart(4, '0');
      else out += c;
    }
    return out + '"';
  }

  const result = serialize(value, '', 0);
  return result;   // 顶层若是 undefined 就返回 undefined，与原生一致
}

关键点：

  • undefined / function / symbol 在对象里被跳过，在数组里变 null，顶层直接返回 undefined。
  • NaN / Infinity 变 null。
  • toJSON 方法：如果值上有 toJSON 方法（如 Date 的 toJSON 返回 ISO 字符串），先调用它拿到序列化用值。
  • 循环引用 用 WeakSet 记录，命中就抛 TypeError（跟原生一致）。
  • BigInt 抛错。
  • space 参数支持缩进（数字或字符串），生成漂亮 JSON。
  • replacer 函数 应用到每个键值对。

myParse ​

js
function myParse(text) {
  let i = 0;
  const str = String(text);

  function parseValue() {
    skipWhitespace();
    const ch = str[i];
    if (ch === '{') return parseObject();
    if (ch === '[') return parseArray();
    if (ch === '"') return parseString();
    if (ch === 't' || ch === 'f') return parseBool();
    if (ch === 'n') return parseNull();
    return parseNumber();
  }

  function parseObject() {
    i++;   // '{'
    const obj = {};
    skipWhitespace();
    if (str[i] === '}') { i++; return obj; }
    while (i < str.length) {
      skipWhitespace();
      if (str[i] !== '"') throw new SyntaxError(`Expected " at ${i}`);
      const key = parseString();
      skipWhitespace();
      if (str[i] !== ':') throw new SyntaxError(`Expected : at ${i}`);
      i++;
      obj[key] = parseValue();
      skipWhitespace();
      if (str[i] === ',') { i++; continue; }
      if (str[i] === '}') { i++; return obj; }
      throw new SyntaxError(`Unexpected ${str[i]} at ${i}`);
    }
    throw new SyntaxError('Unterminated object');
  }

  function parseArray() {
    i++;   // '['
    const arr = [];
    skipWhitespace();
    if (str[i] === ']') { i++; return arr; }
    while (i < str.length) {
      arr.push(parseValue());
      skipWhitespace();
      if (str[i] === ',') { i++; continue; }
      if (str[i] === ']') { i++; return arr; }
      throw new SyntaxError(`Unexpected ${str[i]} at ${i}`);
    }
    throw new SyntaxError('Unterminated array');
  }

  function parseString() {
    i++;   // 开头 "
    let out = '';
    while (i < str.length && str[i] !== '"') {
      if (str[i] === '\\') {
        i++;
        const esc = str[i++];
        if (esc === 'u') {
          const hex = str.slice(i, i + 4);
          out += String.fromCharCode(parseInt(hex, 16));
          i += 4;
        } else {
          const map = { '"': '"', '\\': '\\', '/': '/', n: '\n', r: '\r', t: '\t', b: '\b', f: '\f' };
          if (esc in map) out += map[esc];
          else throw new SyntaxError(`Bad escape \\${esc} at ${i}`);
        }
      } else {
        out += str[i++];
      }
    }
    if (str[i] !== '"') throw new SyntaxError('Unterminated string');
    i++;
    return out;
  }

  function parseNumber() {
    const start = i;
    if (str[i] === '-') i++;
    while (i < str.length && /[0-9.eE+\-]/.test(str[i])) i++;
    const numStr = str.slice(start, i);
    const num = Number(numStr);
    if (Number.isNaN(num)) throw new SyntaxError(`Bad number ${numStr} at ${start}`);
    return num;
  }

  function parseBool() {
    if (str.slice(i, i + 4) === 'true')  { i += 4; return true; }
    if (str.slice(i, i + 5) === 'false') { i += 5; return false; }
    throw new SyntaxError(`Bad boolean at ${i}`);
  }

  function parseNull() {
    if (str.slice(i, i + 4) === 'null') { i += 4; return null; }
    throw new SyntaxError(`Bad null at ${i}`);
  }

  function skipWhitespace() {
    while (i < str.length && /\s/.test(str[i])) i++;
  }

  const value = parseValue();
  skipWhitespace();
  if (i < str.length) throw new SyntaxError(`Unexpected data after JSON at ${i}`);
  return value;
}

追问 1：JSON.stringify 的 replacer 和 reviver 是干什么的？ ​

答：

replacer（stringify 第二个参数）：

  • 函数：function (key, value) 会在序列化每一对键值时被调用，返回值作为最终序列化的值。返回 undefined 相当于跳过这个键。可以用来过滤敏感字段、做类型转换。
    js
    JSON.stringify(user, (k, v) => k === 'password' ? undefined : v);
    1
  • 数组：作为白名单，只序列化数组里列出的 key。
    js
    JSON.stringify(user, ['name', 'email']);   // 只保留这两个字段
    1

reviver（parse 第二个参数）：

  • 反序列化时对每个键值对做后处理。常用于把字符串转成 Date、修正精度、过滤字段。
    js
    JSON.parse(s, (k, v) => k === 'date' ? new Date(v) : v);
    1

两者搭配可以做"自定义序列化协议"，比如让 Map/Set 也能 round-trip：

js
const s = JSON.stringify(map, (k, v) =>
  v instanceof Map ? { __type: 'Map', entries: [...v] } : v);
const back = JSON.parse(s, (k, v) =>
  v && v.__type === 'Map' ? new Map(v.entries) : v);

追问 2：JSON.stringify 序列化时对象属性的顺序是怎样的？ ​

答：跟 Object.keys 的顺序一致，遵循以下规则：

  1. 整数键（"0", "1", "2"...） 先按数字升序；
  2. 其他字符串键 按插入顺序；
  3. Symbol 键 不参与序列化（JSON.stringify 直接忽略）。
js
JSON.stringify({ b: 1, 1: 'x', a: 2, 0: 'y' });
// '{"0":"y","1":"x","b":1,"a":2}'
1
2

这个顺序在 ES2015 起被规范化，之前是 engine-specific 的。业务里若需要"稳定的字段顺序"（比如做数据签名、hash 校验），应该：

  1. 手动构造有序对象，或者
  2. 用 replacer 强制排序：JSON.stringify(obj, Object.keys(obj).sort())；
  3. 或者用第三方库 fast-json-stable-stringify，它对所有嵌套层都排序。

追问 3：JSON 序列化中有哪些常见的坑？ ​

答：

1. Date 变字符串

js
JSON.stringify({ d: new Date() });
// '{"d":"2024-01-01T00:00:00.000Z"}'
JSON.parse('{"d":"2024-..."}').d;   // 是字符串，不是 Date！
1
2
3

要恢复 Date，用 reviver 检测 ISO 字符串手动 new。

2. undefined 消失、null 保留

js
JSON.stringify({ a: undefined, b: null });   // '{"b":null}'
1

3. NaN / Infinity 变 null

js
JSON.stringify({ x: NaN });   // '{"x":null}'
1

4. Map / Set 变空对象

js
JSON.stringify(new Map([['k','v']]));   // '{}'
JSON.stringify(new Set([1,2]));         // '{}'
1
2

5. BigInt 抛 TypeError

js
JSON.stringify(1n);   // TypeError
1

需要自定义处理：字段挂 toJSON 或者 stringify 时用 replacer 转字符串。

6. 循环引用抛 TypeError

js
const a = {}; a.self = a;
JSON.stringify(a);   // TypeError: Converting circular structure to JSON
1
2

7. 大整数精度丢失

后端 Java Long 超过 Number.MAX_SAFE_INTEGER，JSON.parse 拿到的 number 已经不精确。用 json-bigint 之类的库处理。

8. 数据里的 <、>、&、\u2028、\u2029

若序列化后要嵌进 HTML 或 JS，需要额外转义 <、>、\u2028（Line Separator）等字符，防 XSS 和 JS 解析错误。库如 serialize-javascript 做这类安全序列化。
