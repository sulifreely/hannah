题目 ​

js
class TreeNode {
  constructor(val = 0, left = null, right = null) {
    this.val = val;
    this.left = left;
    this.right = right;
  }
}

实现以下遍历函数，分别给出递归版和迭代版：

  1. 前序遍历（根-左-右）：preorder(root) 返回节点值的数组
  2. 中序遍历（左-根-右）：inorder(root) 返回节点值的数组
  3. 后序遍历（左-右-根）：postorder(root) 返回节点值的数组

另外实现： 4. 层序遍历（BFS，按层输出）：levelOrder(root) 返回二维数组，每一层是一个子数组

迭代版要求用统一的颜色标记法（或栈模拟法），三种遍历用同一种思想实现。

参考答案 ​

递归版（简单直观） ​

js
// 前序：根 -> 左 -> 右
function preorderRecursive(root) {
  const res = [];
  function dfs(node) {
    if (!node) return;
    res.push(node.val);     // 根
    dfs(node.left);         // 左
    dfs(node.right);        // 右
  }
  dfs(root);
  return res;
}

// 中序：左 -> 根 -> 右
function inorderRecursive(root) {
  const res = [];
  function dfs(node) {
    if (!node) return;
    dfs(node.left);
    res.push(node.val);
    dfs(node.right);
  }
  dfs(root);
  return res;
}

// 后序：左 -> 右 -> 根
function postorderRecursive(root) {
  const res = [];
  function dfs(node) {
    if (!node) return;
    dfs(node.left);
    dfs(node.right);
    res.push(node.val);
  }
  dfs(root);
  return res;
}

复杂度：时间 O(n)，每个节点访问一次；空间 O(h)，h 是树高（递归栈深度，最坏 O(n)，平均 O(log n)）。

迭代版（颜色标记法 / 统一写法） ​

核心思想：栈中元素用 [node, color] 表示，color 为 0 表示「未访问」，color 为 1 表示「已访问」。

  • 遇到未访问的节点：标记为已访问，然后把右节点、左节点、自身按顺序重新入栈（栈是 LIFO，所以要倒序入栈）。
  • 遇到已访问的节点：输出它的值。

入栈顺序决定遍历顺序：

  • 前序（根-左-右）：右、左、根（根最后入栈，最先弹出输出）
  • 中序（左-根-右）：右、根、左
  • 后序（左-右-根）：根、右、左
js
// 前序遍历：根-左-右
function preorderIterative(root) {
  if (!root) return [];
  const res = [];
  const stack = [[root, 0]]; // [节点, 颜色：0=未访问, 1=已访问]
  
  while (stack.length) {
    const [node, color] = stack.pop();
    if (color === 1) {
      res.push(node.val);
    } else {
      // 入栈顺序：右 -> 左 -> 根（弹出顺序才是根 -> 左 -> 右）
      if (node.right) stack.push([node.right, 0]);
      if (node.left) stack.push([node.left, 0]);
      stack.push([node, 1]); // 标记为已访问，下次弹出直接输出
    }
  }
  return res;
}

// 中序遍历：左-根-右
function inorderIterative(root) {
  if (!root) return [];
  const res = [];
  const stack = [[root, 0]];
  
  while (stack.length) {
    const [node, color] = stack.pop();
    if (color === 1) {
      res.push(node.val);
    } else {
      // 入栈顺序：右 -> 根 -> 左
      if (node.right) stack.push([node.right, 0]);
      stack.push([node, 1]);
      if (node.left) stack.push([node.left, 0]);
    }
  }
  return res;
}

// 后序遍历：左-右-根
function postorderIterative(root) {
  if (!root) return [];
  const res = [];
  const stack = [[root, 0]];
  
  while (stack.length) {
    const [node, color] = stack.pop();
    if (color === 1) {
      res.push(node.val);
    } else {
      // 入栈顺序：根 -> 右 -> 左
      stack.push([node, 1]);
      if (node.right) stack.push([node.right, 0]);
      if (node.left) stack.push([node.left, 0]);
    }
  }
  return res;
}

记忆技巧：「根」放在哪决定了遍历类型——

  • 前序：根最后压入（最先弹输出）
  • 中序：根中间压入
  • 后序：根最先压入（最后弹输出）

层序遍历（BFS） ​

js
function levelOrder(root) {
  if (!root) return [];
  const res = [];
  const queue = [root];
  
  while (queue.length) {
    const levelSize = queue.length; // 当前层的节点数
    const level = [];
    
    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift(); // 出队
      level.push(node.val);
      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }
    
    res.push(level);
  }
  
  return res;
}

思路：

  • 用队列存当前层的所有节点。
  • 每轮开始时先记录当前层节点数 levelSize，然后一次性处理完当前层所有节点（避免混入下一层）。
  • 处理当前层时把下一层节点入队。

复杂度：时间 O(n)，每个节点入队出队各一次；空间 O(n)，队列最大宽度可能是最后一层节点数（完全二叉树最后一层约 n/2）。

追问 1：二叉树的Morris 遍历听说过吗？它是怎么做到 O(1) 空间的？ ​

答 ​

Morris 遍历（莫里斯遍历）是一种不需要栈、不需要递归的二叉树遍历方法，空间复杂度 O(1)，时间 O(n)。

核心思想：利用叶子节点的大量空右指针，让它们临时指向「中序后继」（也就是遍历序列中的下一个节点），遍历完左子树后通过这个指针回到父节点，再把指针恢复。

以中序遍历为例：

js
function morrisInorder(root) {
  const res = [];
  let curr = root;
  
  while (curr) {
    if (!curr.left) {
      // 没有左孩子，直接输出，然后去右孩子
      res.push(curr.val);
      curr = curr.right;
    } else {
      // 找左子树的最右节点（中序前驱）
      let pred = curr.left;
      while (pred.right && pred.right !== curr) {
        pred = pred.right;
      }
      
      if (!pred.right) {
        // 第一次到达：建立线索（临时指向 curr）
        pred.right = curr;
        curr = curr.left;
      } else {
        // 第二次到达：说明左子树已遍历完，输出 curr，恢复线索
        pred.right = null;
        res.push(curr.val);
        curr = curr.right;
      }
    }
  }
  
  return res;
}

追问 2：N 叉树的前序、后序、层序遍历怎么写？和二叉树有什么区别？ ​

答 ​

N 叉树节点定义：

js
class NNode {
  constructor(val, children = []) {
    this.val = val;
    this.children = children; // 子节点数组
  }
}

核心区别：不再是 left/right 两个孩子，而是一个 children 数组。递归和迭代的思路都一样，只是从「两个方向」变成「遍历 children 数组」。

js
// 前序（根 -> 孩子们）
function naryPreorder(root) {
  if (!root) return [];
  const res = [];
  const stack = [root];
  while (stack.length) {
    const node = stack.pop();
    res.push(node.val);
    // 注意：栈是 LIFO，children 要倒序入栈，才能保证弹出顺序和 children 顺序一致
    for (let i = node.children.length - 1; i >= 0; i--) {
      stack.push(node.children[i]);
    }
  }
  return res;
}

// 后序（孩子们 -> 根）
function naryPostorder(root) {
  if (!root) return [];
  const res = [];
  const stack = [[root, 0]]; // 颜色标记法
  while (stack.length) {
    const [node, color] = stack.pop();
    if (color === 1) {
      res.push(node.val);
    } else {
      stack.push([node, 1]);
      // 倒序入栈，保证弹出顺序是 children[0] 到 children[n-1]
      for (let i = node.children.length - 1; i >= 0; i--) {
        stack.push([node.children[i], 0]);
      }
    }
  }
  return res;
}

// 层序（BFS）
function naryLevelOrder(root) {
  if (!root) return [];
  const res = [];
  const queue = [root];
  while (queue.length) {
    const size = queue.length;
    const level = [];
    for (let i = 0; i < size; i++) {
      const node = queue.shift();
      level.push(node.val);
      queue.push(...node.children); // 所有孩子入队
    }
    res.push(level);
  }
  return res;
}

追问 3：前序 + 中序遍历结果可以唯一确定一棵二叉树，那后序 + 中序呢？前序 + 后序呢？为什么？ ​

答 ​

结论：

  • ✅ 前序 + 中序 → 可以唯一确定
  • ✅ 后序 + 中序 → 可以唯一确定
  • ❌ 前序 + 后序 → 不能唯一确定（除非是真二叉树，即所有非叶子节点都有两个孩子）

为什么需要中序？

  • 前序 / 后序只能告诉你「谁是根」，但无法区分左子树和右子树的边界。
  • 中序告诉你「根左边都是左子树，右边都是右子树」，有了这个划分才能递归构造下去。

前序 + 中序 构造二叉树的思路：

js
function buildTree(preorder, inorder) {
  if (!preorder.length) return null;
  
  const rootVal = preorder[0]; // 前序第一个是根
  const root = new TreeNode(rootVal);
  
  const mid = inorder.indexOf(rootVal); // 根在中序中的位置
  
  // 左子树：前序 [1..mid]，中序 [0..mid-1]
  root.left = buildTree(preorder.slice(1, mid + 1), inorder.slice(0, mid));
  // 右子树：前序 [mid+1..end]，中序 [mid+1..end]
  root.right = buildTree(preorder.slice(mid + 1), inorder.slice(mid + 1));
  
  return root;
}

为什么前序 + 后序不行？ 举个反例：

 /           \

  • 前序都是 [1, 2]
  • 后序都是 [2, 1]

两棵不同的树，前序和后序完全一样，所以无法唯一确定。只有当所有非叶子节点都恰好有两个孩子时（真二叉树），前序 + 后序才能唯一确定一棵树。

面试常考拓展：

  • 中序 + 后序 构造 → 后序最后一个是根
  • 序列化 / 反序列化 → 前序 + 空节点标记（#）也能唯一确定（因为空节点提供了边界信息）
